fade-in {
    # メッセージボックスをアニメーションする
    layer: msg;

    # 開始から0.5秒かける
    start: 0.0;
    end: 0.5;

    # 位置は変更しない
    from-x: +0;
    from-y: +0;
    to-x: +0;
    to-y: +0;

    # 透明度を0から255にする
    from-a: 0;
    to-a: 255;

    # メッセージボックスを表示する
    show-special: yes;
}
